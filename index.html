Класс Автомобиля на C++

Задача: Создать класс автомобиля с наследованием и свойствами

Код
.cpp
#include <iostream>
using namespace std;

class Car {
public:
    string model;
    int wheels;
    string doors;
    string headlights;
    string hood;

    Car(string m) {
        model = m;
        wheels = 4;
        doors = "Standard";
        headlights = "LED";
        hood = "Metal";
    }

    virtual void displayInfo() {
        cout << "Model: " << model << "\n";
    }
};

class CarModel1 : public Car {
public:
    CarModel1(): Car("Car Model 1") {}
};

class CarModel2 : public Car {
public:
    CarModel2(): Car("Car Model 2") {}
};

class CarModel3 : public Car {
public:
    CarModel3(): Car("Car Model 3") {}
};

int main() {
    CarModel1 car1;
    CarModel2 car2;
    CarModel3 car3;

    car1.displayInfo();
    car2.displayInfo();
    car3.displayInfo();

    return 0;
}
