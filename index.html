Класс Автомобиль на C++

Задача: Создать класс автомобиля с наследованием, свойствами и методами движения, остановки и старта

Код
.cpp
#include <iostream>
using namespace std;

// Базовый класс Car
class Car {
public:
    string model;
    int wheels;
    string doors;
    string headlights;
    string hood;
    bool isMoving;

    Car(string m) {
        model = m;
        wheels = 4;
        doors = "Standard";
        headlights = "LED";
        hood = "Metal";
        isMoving = false;
    }

    virtual void displayInfo() {
        cout << "Model: " << model << "\n";
    }

    // Метод для запуска движения
    void start() {
        if (!isMoving) {
            isMoving = true;
            cout << model << " started moving.\n";
        } else {
            cout << model << " is already moving.\n";
        }
    }

    // Метод для остановки движения
    void stop() {
        if (isMoving) {
            isMoving = false;
            cout << model << " stopped.\n";
        } else {
            cout << model << " is already stopped.\n";
        }
    }

    // Метод для движения вперёд
    void move() {
        if (isMoving) {
            cout << model << " is moving forward.\n";
        } else {
            cout << model << " is not moving. Start the car first.\n";
        }
    }
};

// Модель автомобиля 1
class CarModel1 : public Car {
public:
    CarModel1(): Car("Car Model 1") {}
};

// Модель автомобиля 2
class CarModel2 : public Car {
public:
    CarModel2(): Car("Car Model 2") {}
};

// Модель автомобиля 3
class CarModel3 : public Car {
public:
    CarModel3(): Car("Car Model 3") {}
};

int main() {
    CarModel1 car1;
    CarModel2 car2;
    CarModel3 car3;

    // Тестирование методов для car1
    car1.displayInfo();
    car1.start();
    car1.move();
    car1.stop();

    // Тестирование методов для car2
    car2.displayInfo();
    car2.start();
    car2.move();
    car2.stop();

    // Тестирование методов для car3
    car3.displayInfo();
    car3.start();
    car3.move();
    car3.stop();

    return 0;
}
